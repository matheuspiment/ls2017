19  /**
20   * Uma regra define como avaliar um ou mais
21   * objetos avaliáveis. Um objeto é avaliável
22   * se implementa a interface {@link Avaliavel}).
23   *
24   * <p>Ao ser executada uma regra produz um valor
25   * que se torna disponível por meio da variável
26   * definida para a regra. Por exemplo, uma regra
27   * definida pela expressão "x + y", quando avaliada,
28   * resulta no valor dessa soma associada à variável
29   * definida para a regra.
30   *
31   * <p>A avaliação de uma regra pode dependenter de
32   * valores devidamente identificados em um contexto.
33   * Quando a regra é avaliada, a "variável" correspondente
34   * passa a existir no contexto e, dessa forma, torna-se
35   * disponível para outras regras.
36   *
37   * <p>Uma regra só é avaliada após a avaliação das
38   * regras das quais depende. A dependência entre
39   * regras é estabelecida por meio de suas variáveis.
40   * Por exemplo, uma regra que possui como variável
41   * o identificador "x" só deve ser avaliada antes de
42   * qualquer regra que depende do identificador "x".
43   *
44   * <p>Uma regra pode ser empregada para obter a média
45   * de pontos obtidos com o ensino em determinado período.
46   * Nesse caso, não se trata de uma simples contagem ou
47   * consulta a propriedades de relatos. A regra em questão
48   * é aplicada sobre um conjunto de entrada no qual cada
49   * elemento possui um atributo devidamente identificado,
50   * sobre o qual a média será calculada.
51   *
52   * <p>O resultado da avaliação de uma regra é depositada
53   * em uma variável. Em consequência, para um determinado
54   * conjunto de regras, o identificador da variável que
55   * irá reter o valor final de uma regra deve ser única
56   * nesse conjunto.
57   */
58  public abstract class Regra {
59  
60      /**
61       * Descrição da regra.
62       */
63      private String descricao;
64  
65      /**
66       * Valor máximo admitido para a avaliação da regra.
67       * Se a avaliação de uma regra produzir um valor
68       * acima do valor indicado por essa propriedade,
69       * então o valor produzido é substituído pelo
70       * valor da propriedade.
71       */
72      private float valorMaximo = Float.MAX_VALUE;
73  
74      /**
75       * Valor mínimo aditimido para o resultado
76       * da avaliação da regra. Se um valor inferior
77       * é obtido, então o valor resultante é
78       * substituído pelo valor dessa propriedade.
79       */
80      private float valorMinimo = Float.MIN_VALUE;
81  
82      /**
83       * Nome da variável (atributo) que guardará
84       * o resultado da avaliação da regra.
85       *
86       * <p>Trata-se de chave natural para um dado
87       * conjunto de regras. Em tempo, uma regra só
88       * produz um único valor.
89       */
90      private String variavel;
91  
92      /**
93       * Lista de identificadores de atributos que são
94       * empregados pela expressão que avalia a regra.
95       * Caso a regra seja condicional, então acumula
96       * os identificados das expressões "então" e
97       * "senão". Se a regra é do tipoRelato pontos por item
98       * avaliável, então a lista é vazia.
99       *
100      * <p>Convém observar que uma regra também inclui
101      * identificadores de funções que, se forem empregadas,
102      * também fazem parte dessa lista.
103      */
104     private List<String> dependeDe;
105 
106     /**
107      * Identificador da classe que restringe os relatos a
108      * serem empregados na avaliação da regra.
109      */
110     private String classe;
111 
112     /**
113      * Conjunto de funções empregadas pela regra.
114      */
115     private List<Funcao> funcoes;
116 
117     /**
118      * Representa lista vazia de funções.
119      */
120     private final static List<Funcao> vazia = new ArrayList<>(0);
121 
122     /**
123      * Obtém a lista de funções associadas à regra.
124      *
125      * @return Conjunto de funções empregado pela regra.
126      */
127     public List<Funcao> getFuncoes() {
128         return funcoes;
129     }
130 
131     /**
132      * Cria uma regra.
133      *
134      * @param resultado      O identificador (nome) da variável que retém o
135      *                      valor da avaliação da regra. Em um dado conjunto de
136      *                      regras, existe uma variável distinta para cada uma
137      *                      delas.
138      * @param detalhes     Texto que fornece alguma explanação sobre a regra.
139      * @param maximo   O valor máximo a ser utilizado como resultado da
140      *                      avaliação da regra. Esse valor é empregado apenas
141      *                      se a avaliação resultar em valor superior ao
142      *                      expresso por esse parâmetro.
143      * @param minimo   O valor mínimo a ser utilizado como resultado da
144      *                      avaliação da regra. Esse valor é empregado apenas
145      *                      se a avaliação resultar em valor inferior ao
146      *                      expresso por esse parâmetro.
147      * @throws CampoExigidoNaoFornecido Caso um campo obrigatório para a
148      *                                  definição de uma regra não seja
149      *                                  fornecido.
150      */
151     public Regra(final String resultado,
152                  final String detalhes,
153                  final float maximo,
154                  final float minimo) {
155 
156         if (resultado == null || resultado.isEmpty()) {
157             throw new CampoExigidoNaoFornecido("variavel");
158         }
159 
160         if (detalhes == null || detalhes.isEmpty()) {
161             throw new CampoExigidoNaoFornecido("descricao");
162         }
163 
164         if (minimo > maximo) {
165             throw new IllegalArgumentException("minimo maior que maximo");
166         }
167 
168         this.descricao = detalhes;
169         this.valorMaximo = maximo;
170         this.valorMinimo = minimo;
171         this.variavel = resultado;
172         this.funcoes = vazia;
173     }
174 
175     /**
176      * Cria uma regra.
177      *
178      * @param resultado      O identificador (nome) da variável que retém o
179      *                      valor da avaliação da regra. Em um dado conjunto de
180      *                      regras, existe uma variável distinta para cada uma
181      *                      delas.
182      * @param detalhes     Texto que fornece alguma explanação sobre a regra.
183      * @param maximo   O valor máximo a ser utilizado como resultado da
184      *                      avaliação da regra. Esse valor é empregado apenas
185      *                      se a avaliação resultar em valor superior ao
186      *                      expresso por esse parâmetro.
187      * @param minimo   O valor mínimo a ser utilizado como resultado da
188      *                      avaliação da regra. Esse valor é empregado apenas
189      *                      se a avaliação resultar em valor inferior ao
190      *                      expresso por esse parâmetro.
191      * @param classe A classe que restringe os relatos a serem empregados
192      *               na avaliação da regra.
193      *
194      * @throws CampoExigidoNaoFornecido Caso um campo obrigatório para a
195      *                                  definição de uma regra não seja
196      *                                  fornecido.
197      */
198     public Regra(final String resultado,
199                  final String detalhes,
200                  final float maximo,
201                  final float minimo,
202                  final String classe) {
203         this(resultado, detalhes, maximo, minimo);
204 
205         this.classe = classe;
206         this.funcoes = vazia;
207     }
208 
209     /**
210      * Cria uma regra.
211      *
212      * @param resultado      O identificador (nome) da variável que retém o
213      *                      valor da avaliação da regra. Em um dado conjunto de
214      *                      regras, existe uma variável distinta para cada uma
215      *                      delas.
216      * @param detalhes     Texto que fornece alguma explanação sobre a regra.
217      * @param maximo   O valor máximo a ser utilizado como resultado da
218      *                      avaliação da regra. Esse valor é empregado apenas
219      *                      se a avaliação resultar em valor superior ao
220      *                      expresso por esse parâmetro.
221      * @param minimo   O valor mínimo a ser utilizado como resultado da
222      *                      avaliação da regra. Esse valor é empregado apenas
223      *                      se a avaliação resultar em valor inferior ao
224      *                      expresso por esse parâmetro.
225      * @param classe A classe que restringe os relatos a serem empregados
226      *               na avaliação da regra.
227      *
228      * @param funcoes O conjunto de funções empregadas pela regra.
229      * @throws CampoExigidoNaoFornecido Caso um campo obrigatório para a
230      *                                  definição de uma regra não seja
231      *                                  fornecido.
232      */
233     public Regra(final String resultado,
234                  final String detalhes,
235                  final float maximo,
236                  final float minimo,
237                  final String classe,
238                  final List<Funcao> funcoes) {
239         this(resultado, detalhes, maximo, minimo);
240 
241         this.classe = classe;
242         this.funcoes = funcoes == null ? vazia : funcoes;
243     }
244 
245     public Regra() {
246         valorMaximo = Float.MAX_VALUE;
247         valorMinimo = Float.MIN_VALUE;
248     }

270      * Recupera o valor mínimo admitido pela regra.

302 
303     /**
304      * Lista de dependeDe diretamente empregados
305      * pela expressão cuja avaliação dá origem à
306      * pontuação da regra.
307      *
308      * @return Lista de dependeDe diretamente empregados
309      *      para avaliação da regra.
310      */
311     public final List<String> getDependeDe() {
312         return dependeDe;
313     }

315     /**
316      * Define a lista de variáveis das quais a presente
317      * regra depende.

319      * @param dependencias Lista de identificadores de
320      *                     variáveis dos quais a presente
321      *                     regra depende.
322      */
323     public final void setDependeDe(final List<String> dependencias) {
324         dependeDe = dependencias;
325     }

327     @Override
328     public boolean equals(final Object outro) {
329         if (this == outro) {
330             return true;
331         }
332 
333         if (outro == null || getClass() != outro.getClass()) {
334             return false;
335         }
336 
337         Regra regra = (Regra) outro;
338 
339         return variavel.equals(regra.variavel);
340     }
341 
342     @Override
343     public int hashCode() {
344         return variavel.hashCode();
345     }

347     /**
348      * A avaliação de uma regra produz um valor, que pode
349      * estar fora dos limites estabelecidos pela regra.
350      * Esse método ajusta um valor conforme os limites
351      * estabelecidos pela regra.
352      *
353      * <p>Um valor abaixo do limite inferior é corrigido
354      * para o limite inferior e, um valor acima do limte
355      * superior, é corrigido para o limite superior.
356      *
357      * @param valor Valor a ser ajustado.
358      *
359      * @return O valor fornecido, caso esteja dentro dos limites
360      * estabelecidos pela regra ou o valor resultante da
361      * aplicação dos limites da regra.
362      */
363     public final float ajustaLimites(final float valor) {
364         if (valor < getValorMinimo()) {
365             return getValorMinimo();
366         }
367 
368         if (valor > getValorMaximo()) {
369             return getValorMaximo();
370         }
371 
372         return valor;
373     }

387     }

389     /**
390      * Avalia a regra, produzindo um valor a ser associado
391      * à variável definida para a regra.
392      *
393      * <p>A avaliação de uma regra depende de valores de itens
394      * avaliáveis e/ou do valor de variáveis. Em tempo, as
395      * variáveis são definidas pela avaliação de regras.
396      *
397      * <p>Uma regra deve ser chamada apenas após a avaliação
398      * de regras que produzem valores para as variáveis das
399      * quais depende. Caso contrário, valores padrão serão
400      * empregados.
401      *
402      * @param avaliaveis Conjunto de itens a serem avaliados.
403      *
404      * @param contexto Valores de variáveis que podem ser
405      *                 necessárias para avaliação da regra.
406      *
407      * @return Resultado da avaliação da regra.
408      */
409     public abstract Valor avalie(List<? extends Avaliavel> avaliaveis,
410                                  Map<String, Valor> contexto);
