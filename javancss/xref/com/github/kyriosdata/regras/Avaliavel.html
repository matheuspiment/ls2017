7   package com.github.kyriosdata.regras;
8   
9   import com.github.kyriosdata.regras.regra.Regra;
10  
11  /**
12   * Representa qualquer item que pode ser
13   * avaliado por uma regra. Adicionalmente,
14   * um avaliável é um item que pode ser
15   * "modificado" por uma observação.
16   *
17   * @see Observacao
18   * @see Regra
19   * @see Relato
20   */
21  public interface Avaliavel {
22  
23      /**
24       * Recupera o valor do atributo para o
25       * objeto em questão.
26       *
27       * @param atributo O identificador único do atributo.
28       *
29       * @return O valor do atributo.
30       */
31      Valor get(String atributo);
32  
33      /**
34       * Classe associada ao item avaliável.
35       *
36       * @return Identificador da classe do avaliável.
37       */
38      String getClasse();
39  }
