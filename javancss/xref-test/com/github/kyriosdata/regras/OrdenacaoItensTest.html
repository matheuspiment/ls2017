40      @Test
41      public void umItemDependenteDeAtributoOrdenacaoTrivial() {
42  
43          // Ensino de graduação: x => rx (usa "x" resultado em "rx")
44          // Ensino a distância: y => ry (usa "y" resultado em "ry")
45          // Ensino: rz = rx + ry (usa "rx" e "ry" resultado em "rz")
46  
47          // Regra: "x" (simples propriedade)
48          List<String> atributosX = new ArrayList<>(1);
49          atributosX.add("x");
50          Regra regraX = new RegraExpressao("rx", "x", 0, 0, "x");
51  
52          // Regra: "y" (simples propriedade)
53          List<String> atributosY = new ArrayList<>(1);
54          atributosY.add("y");
55          Regra regraY = new RegraExpressao("ry", "y", 0, 0, "y");
56  
57          // Regra: "rx + ry" (depende de itens avaliados)
58          // (não consome simples propriedades, mas resultados
59          // da avaliação de itens)
60          List<String> atributosZ = new ArrayList<>();
61          atributosZ.add("rx");
62          atributosZ.add("ry");
63          Regra regraZ = new RegraExpressao("rz", "z", 0, 0, "rx + ry");
64  
65          List<Regra> itens = new ArrayList<>(1);
66          itens.add(regraZ);
67          itens.add(regraX);
68          itens.add(regraY);
69  
70          // Preparação das regras
71          ParserTeste ptX = new ParserTeste();
72          ptX.setDependencias(atributosX);
73  
74          ParserTeste ptY = new ParserTeste();
75          ptY.setDependencias(atributosY);
76  
77          ParserTeste ptZ = new ParserTeste();
78          ptZ.setDependencias(atributosZ);
79  
80          regraX.preparacao(ptX);
81          regraY.preparacao(ptY);
82          regraZ.preparacao(ptZ);
83  
84          List<Regra> ordenados = OrdenacaoService.ordena(itens);
85  
86          assertEquals(3, ordenados.size());
87          assertEquals("rz", ordenados.get(2).getVariavel());
88      }

90      @Test
91      public void duasRegrasOrdenacaoUnica() {
92  
93          // a = 10
94          Regra ra = new RegraExpressao("a", "a = 10", 10, 0, "10");
95  
96          // b = a + 1
97          Regra rb = new RegraExpressao("b", "b = a + 1", 11, 0, "a + 1");
98  
99          // Dependências de "b", apenas "a"
100         List<String> depsB = new ArrayList<>(0);
101         depsB.add("a");
102 
103         // Itens que devem ser ordenados (itemA e itemB)
104         List<Regra> regras = new ArrayList<>(2);
105         regras.add(ra);
106         regras.add(rb);
107 
108         // Preparação das regras (antes de qualquer uso)
109         ParserTeste ptB = new ParserTeste();
110         ptB.setDependencias(depsB);
111 
112         ParserTeste ptA = new ParserTeste();
113         ptA.setDependencias(new ArrayList<>(0));
114 
115         ra.preparacao(ptA);
116         rb.preparacao(ptB);
117 
118         // Executa operação de ordenação
119         List<Regra> ordenados = OrdenacaoService.ordena(regras);
120 
121         // Verifica que itemA precede itemB
122         assertEquals(2, ordenados.size());
123         assertEquals("a", ordenados.get(0).getVariavel());
124         assertEquals("b", ordenados.get(1).getVariavel());
125     }
