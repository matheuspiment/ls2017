11  public class ObservacaoTest {
12  
13      @Test
14      public void semNovoImplicaRemocao() {
15          Avaliavel o = new Pontuacao("o", new Valor("o"));
16          Observacao ok = new Observacao(o, null, "ok");
17          assertTrue(ok.isRemocao());
18      }
19  
20      @Test
21      public void origemDestinoPeloMenosUm() {
22          assertThrows(CampoExigidoNaoFornecido.class, () -> new Observacao(null, null,"ok"));
23      }
24  
25      @Test
26      public void montaRecuperaCorretamenteUmaNota() {
27          Avaliavel o = new Pontuacao("o", new Valor("o"));
28          Avaliavel s = new Pontuacao("s", new Valor("o"));
29  
30          assertNotEquals(o, s);
31  
32          Observacao n = new Observacao(o, s, "simples erro");
33          assertEquals(o, n.getItemOriginal());
34          assertEquals(s, n.getItemNovo());
35          assertEquals("simples erro", n.getJustificativa());
36      }
37  
38      @Test
39      public void origemSignificaInsercao() {
40          Avaliavel o = new Pontuacao("o", new Valor("o"));
41          assertTrue(new Observacao(null, o, "simples erro").isInsercao());
42      }
43  
44      @Test
45      public void destinoPodeSerNull() {
46          Avaliavel o = new Pontuacao("o", new Valor("o"));
47          new Observacao(o, null, "motivo");
48      }
49  
50      @Test
51      public void justificativaNullGeraExcecao() {
52          Avaliavel o = new Pontuacao("o", new Valor("o"));
53          assertThrows(CampoExigidoNaoFornecido.class, () -> new Observacao(o, o, null));
54      }
55  
56      @Test
57      public void tentativaDeCriarNotaComAvaliaveisDeTiposDistintos() {
58          Avaliavel o = new Pontuacao("o", new Valor("o"));
59          HashMap<String, Valor> valores = new HashMap<>(1);
60          valores.put("v", new Valor("v"));
61  
62          Avaliavel d = new Relato("d", valores);
63  
64          assertThrows(AvaliaveisIncompativeis.class, () -> new Observacao(o, d, "tentativa deve falhar"));
65      }
66  }
